📚 Usage

// Install via npm
npm install sync-audio

// Or use via CDN
<script src="https://unpkg.com/sync-audio/dist/index.js"></script>

// Basic usage
const audioSync = new SyncAudio({
  websocketUrl: 'ws://your-server:8080',
  audioFile: 'your-audio.mp3'
});

audioSync.on('ready', ({ isMaster }) => {
  if (isMaster) {
    audioSync.startSync();
  }
});

audioSync.on('bluetooth-latency', ({ latency }) => {
  console.log(`Bluetooth latency: ${latency}ms`);
});
